-- 1. Base de usuários
CREATE TABLE usuarios (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL,
    tipo VARCHAR(20) NOT NULL CHECK (tipo IN ('cliente', 'barbeiro')),
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Cliente
CREATE TABLE clientes (
    id SERIAL PRIMARY KEY,
    usuario_id INT UNIQUE REFERENCES usuarios(id) ON DELETE CASCADE
);

-- 3. Barbeiro
CREATE TABLE barbeiros (
    id SERIAL PRIMARY KEY,
    usuario_id INT UNIQUE REFERENCES usuarios(id) ON DELETE CASCADE,
    barbearia VARCHAR(100),
    endereco TEXT
);

-- 4. Serviços do barbeiro
CREATE TABLE servicos (
    id SERIAL PRIMARY KEY,
    barbeiro_id INT NOT NULL REFERENCES barbeiros(id) ON DELETE CASCADE,
    nome VARCHAR(100) NOT NULL,
    preco NUMERIC(8, 2) NOT NULL,
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 5. Horários disponíveis do barbeiro
CREATE TABLE horarios_disponiveis (
    id SERIAL PRIMARY KEY,
    barbeiro_id INT NOT NULL REFERENCES barbeiros(id) ON DELETE CASCADE,
    dia_semana VARCHAR(15) NOT NULL CHECK (dia_semana IN (
        'segunda', 'terca', 'quarta', 'quinta', 'sexta', 'sabado', 'domingo'
    )),
    hora_inicio TIME NOT NULL,
    hora_fim TIME NOT NULL,
    intervalo_minutos INT NOT NULL DEFAULT 30,
    ativo BOOLEAN DEFAULT TRUE
);

-- 6. Agendamentos
CREATE TABLE agendamentos (
    id SERIAL PRIMARY KEY,
    cliente_id INT NOT NULL REFERENCES clientes(id),
    barbeiro_id INT NOT NULL REFERENCES barbeiros(id),
    servico_id INT NOT NULL REFERENCES servicos(id),
    data_agendamento DATE NOT NULL,
    hora_agendamento TIME NOT NULL,
    status VARCHAR(20) DEFAULT 'pendente' CHECK (status IN (
        'pendente', 'confirmado', 'cancelado', 'concluido'
    )),
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (barbeiro_id, data_agendamento, hora_agendamento)
);

-- 7. Formas de pagamento
CREATE TABLE formas_pagamento (
    id SERIAL PRIMARY KEY,
    agendamento_id INT UNIQUE NOT NULL REFERENCES agendamentos(id) ON DELETE CASCADE,
    metodo VARCHAR(30) CHECK (metodo IN (
        'dinheiro', 'pix', 'cartao_credito', 'cartao_debito'
    )),
    status_pagamento VARCHAR(20) DEFAULT 'pendente' CHECK (status_pagamento IN (
        'pendente', 'pago'
    ))
);
